  72.     /**
  73.      * Insert a GQL Command to the GraphDB instance.
  74.      * @param myGQLQuery the Query to insert
  75.      * @return QueryResult Object if the Response was Successful - null if not!
  76.      */
  77.     public function InsertQuery($myGQLQuery) {
  78.         $query_string = urlencode($myGQLQuery);
  79.         //execute the request
  80.         $result = new GQLRestRequest($this->uri . $query_string, new RestCredentials($this->Username, $this->Password), "application/xml");
  81.         $tem = $result->getResponseBody();
  82.         //build the result
  83.         if ($tem != false) { //if false, the connection refused
  84.             $xml = simplexml_load_string($tem); //parse the response string with simplexml
  85.             if ($xml) {
  86.                 //Meta
  87.                 $queryString = "" . $xml->graphdb->queryresult->query;
  88.                 $queryResult = "" . $xml->graphdb->queryresult->result;
  89.                 $queryDuration = "" . $xml->graphdb->queryresult->duration . "ms";
  90.  
  91.                 if ($xml->graphdb->queryresult->errors->children() != null) {
  92.                     //Errors
  93.                     $queryErrors = $this->readErrors($xml->graphdb->queryresult->errors);
  94.                 }
  95.                 if ($xml->graphdb->queryresult->warnings->children() != null) {
  96.                     //Warnings
  97.                     $queryWarnings = $this->readWarning($xml->graphdb->queryresult->warnings);
  98.                 }
  99.                 if ($xml->graphdb->queryresult->results->children() != null) {
  100.                     //Results
  101.                     $queryVertices = $this->readVertices($xml->graphdb->queryresult->results);
  102.                 }
  103.             }
  104.             return new QueryResult($queryVertices, $queryWarnings, $queryErrors, $queryString, $queryResult, $queryDuration);
  105.         } else {
  106.             return false; //returns false if the connection refused
  107.         }
  108.     }

  171.     private function parseAttribute($myAttributeType, $myAttributeValue) {
  172.  
  173.         if (("Double" == $myAttributeType) || ("Float" == $myAttributeType)) {
  174.             settype($myAttributeValue, "float");
  175.             return $myAttributeValue;
  176.         }
  177.  
  178.         if (("Boolean" == $myAttributeType) || ("Bool" == $myAttributeType)) {
  179.             settype($myAttributeValue, "bool");
  180.             return $myAttributeValue;
  181.         }
  182.  
  183.         if (("Integer" == $myAttributeType) || ("Int64" == $myAttributeType) || ("Int32" == $myAttributeType) || ("UInt64" == $myAttributeType) || ("UInt32" == $myAttributeType)) {
  184.             settype($myAttributeValue, "int");
  185.             return $myAttributeValue;
  186.         }
  187.  
  188.         if ("ObjectUUID" == $myAttributeType) {
  189.             return new ObjectUUID($myAttributeValue);
  190.         }
  191.  
  192.         if ("ObjectRevisionID" == $myAttributeType) {
  193.             return new ObjectRevisionID($myAttributeValue);
  194.         }
  195.  
  196.         if ("DateTime" == $myAttributeType) {
  197.  
  198.             //Hack
  199.             $date = new DateTime($myAttributeValue);
  200.             $date->format("dd.MM.YYYY hh:mm:ss");
  201.             return $date;
  202.         }
  203.  
  204.         return $myAttributeValue;
  205.     }

  207.     private function generateEdgeContent($myEdge) {
  208.  
  209.         $targetVertices = array();
  210.         $i = 0;
  211.         foreach ($myEdge->vertex as $target) {
  212.             $targetVertices["Vertex" . $i] = $this->readVertex($target);

  214.         }

  216.     }

  218.     /**
  219.      * Reads a entity of an vertex
  220.      *
  221.      */
  222.     private function readVertex($myVertex) {
  223.  
  224.         $payLoad = array();
  225.         foreach ($myVertex as $attribute) {
  226.  
  227.             $attributeName = "" . $attribute["name"];
  228.             $attributeType = "" . $attribute["type"];
  229.             $attributeValue = "" . $attribute;
  230.             $payLoad["$attributeName"] = $this->parseAttribute($attributeType, $attributeValue);
  231.         }
  232.  
  233.  
  234.         foreach ($myVertex->edge as $edge) {
  235.  
  236.             $edgeName = "" . $edge["name"];
  237.             $edgeType = "" . $edge["type"];
  238.             $targetVertices = $this->generateEdgeContent($edge);
  239.  
  240.             $tmpEdge = new Edge(null, $targetVertices, null);
  241.             $tmpEdge->setEdgeTypeName($edgeType);
  242.  
  243.  
  244.  
  245.             $payLoad["$edgeName"] = $tmpEdge;
  246.         }
  247.  
  248.  
  249.         $edgelabel = $myVertex->edgelabel;
  250.         if ($edgelabel != null) {
  251.  
  252.             $edgeident = $edgelabel->attribute["name"];
  253.             if ($edgeident != null) {
  254.                 if ($edgeident == "weight") {
  255.                     return new VertexWeightedEdge($payLoad, $edgelabel, $edgelabel->attribute["type"]);
  256.                 } elseif ($edgeident == "group") {
  257.                     return new VertexGroup($payLoad, $this->readVertices($edgelabel));
  258.                 }
  259.             }
  260.         }
  261.  
  262.         return new Vertex($payLoad);
  263.     }
